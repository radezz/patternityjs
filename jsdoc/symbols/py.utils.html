{Boolean} pairCall(fn, pairArgs, context)
Function applies pair arguments to target function. It is possible to overload 'key-value' type function to consume objects as single parameter 
function set(prop, value){
    if(arguments.length === 1 && !pairCall(set, arguments[0])){
        target[prop] = value;
    }
}

//instead of calling 
set('prop', 10);
//you can call
set({
    prop: 10,
    prop2: 20,
    prop3: 30
})
