  4 	 * Scheduler helper function, executes
  5 	 * registered tasks
  6 	 * @private
  7      * @param {Object} tasks
  8 	 */
  9 	function execute(tasks){
 10 		var self = this,
 11 		    now = (new Date()).getTime(),
 12 			task,
 13 			i,
 14 			l;
 15 			
 16 			for(i=0, l=tasks.length; i<l; i++){
 17 				task = tasks[i];
 18 				task.action.call(task.ctx || this, now);
 19 			}
 20 	}

 22 	/**
 23 	 * Scheduler objcet is used to schedule multiple tasks to run
 24 	 * periodically with preset interval.
 25 	 * 
 26 	 * @class Constructor takes interval as argument
 27 	 * @name py.Scheduler
 28 	 * 
 29 	 * @constructor
 30 	 * @param {Number} interval in milliseconds
 31 	 */
 32 	$NS.Class('Scheduler', {
 33 	    
 34 		initialize: function(interval){
 35 			var self = this;
 36 			if(typeof(interval) === 'number' && interval > 0){
 37 				self.__interval = interval;	
 38 				self.__tasks = [];
 39 				self.__isRunning = false;
 40 				self.__handle = null;
 41 			}else{
 42 				throw "interval required";
 43 			}
 44 		},
 45 		
 46 		/**
 47 		 * Function adds task to be scheduled using previously 
 48 		 * set interval.
 49 		 * 
 50 		 * @function
 51 		 * @name py.Scheduler#addTask
 52          * @param {Function} fn function which will be scheduled
 53          * @param {Object} [optional] context which should be applied to task 
 54 		 */
 55 		addTask: function(fn, context){
 56 			if(typeof(fn) === 'function'){
 57 				this.__tasks.push({
 58 					action: fn,
 59 					ctx: context
 60 				});	
 61 			}else{
 62 				throw "taks must be a function";
 63 			}
 64 		},
 65 		
 66 		/**
 67 		 * Function changes interval for the current scheduler instance
 68 		 * 
 69 		 * @function
 70 		 * @name py.Scheduler#interval
 71          * @param {Number} interval in milliseconds
 72 		 */
 73 		interval: function(interval){
 74 		    this.__interval = interval;
 75 		},
 76 		
 77 		/**
 78 		 * Function removes previously added tasks from scheduler.
 79          * 
 80          * @function
 81          * @name py.Scheduler#removeTask
 82          * @param {Function} fn
 83 		 */
 84 		removeTask: function(fn){
 85 			var self = this,
 86 				tasks = self.__tasks,
 87 				i = tasks.length;
 88 			while(i--){
 89 				if(tasks[i].action === fn){
 90 					tasks.splice(i,1);
 91 					return;
 92 				}
 93 			}
 94 		},
 95 		
 96 		/**
 97 		 * Function stops the scheduler and clears all registered tasks.
 98 		 * 
 99 		 * @function
100 		 * @name py.Scheduler#reset
101 		 */
102 		reset: function(){
103 			this.stop();
104 			this.__tasks = [];
105 		},
106 		
107 		/**
108 		 * Function starts scheduler.
109 		 * 
110 		 * @function
111 		 * @name py.Scheduler#run
112          * @param {[Function]} fn (optional) if provided will be added to current tasks list
113 		 */
114 		run: function(fn){
115 			var self = this;
116 			
117 			if(fn){
118 				self.addTask(fn);	
119 			}
120 			
121 			self.__isRunning = true;
122 			self.__handle = setInterval(function(){
123 				if(self.__isRunning){
124 					execute(self.__tasks);
125 				}
126 			},self.__interval);
127 		},
128 		
129 		/**

131 		 * 
132 		 * @function
133 		 * @name py.Scheduler#stop
134 		 * 
135 		 */
136 		stop: function(){
137 			this.__isRunning = false;
138 			clearInterval(this.__handle);
139 		},
140 		
141 		/**
142 		 * Function indicates if the scheduler is currently

144 		 * 
145 		 * @function
146 		 * @name py.Scheduler#isRunning
147 		 * 
148 		 * @returns {Boolean} is running
149 		 */
