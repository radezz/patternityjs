1 (function($NS){
  2 	
  3 	/**
  4 	 * Helper function creates new key for 
  5 	 * listener in registry
  6 	 * @private
  7 	 */
  8 	function createKey(){
  9 		return '' + this.__length++;
 10 	}
 11 	
 12 	/**
 13 	 * Helper function which calls registered function
 14 	 * @private
 15      * @param {Object} registered
 16 	 */
 17 	function callRegistered(registered){
 18 		registered.handler.apply(registered.ctx || this, registered.args);
 19 		registered.clearAspect();
 20 	}
 21 	
 22 	/**
 23 	 * @class Sync is used synchronise callbacks as well as to notify when
 24 	 * all registered callback were called.
 25 	 * 
 26 	 * @name py.Sync
 27 	 * 
 28 	 * @params {Object} settings object which may contain following attributes:
 29 	 * <ul>
 30 	 *     <li><code>synchronize</code> - boolean value if set to true all registered handlers will be called synchronously</li>
 31 	 *     <li><code>onAllReady</code> - function which will be called after all handlers were executed</li>
 32 	 * </ul> 
 33 	 * 
 34 	 * @example
 35 	 * 
 36 	 * var handlers = {
 37 	 *     onJsonLoad: function(xhr){},
 38 	 *     onTemplateLoad: function(template){}
 39 	 * };
 40 	 * 
 41 	 * var sync = new py.Sync({
 42 	 *     onAllReady: function(){
 43 	 *         //json is loaded and template too
 44 	 *     }
 45 	 * });
 46 	 * 
 47 	 * sync.addHandler(handlers, 'onJsonLoad');
 48 	 * sync.addHandler(handlers, 'onTemplateLoad');
 49 	 * 
 50 	 */
 51 	$NS.Class('Sync', {
 52 		
 53 		initialize: function(settings){
 54 			var self = this;
 55 			
 56 			settings = settings || {};
 57 			self.__registry = {};
 58 			self.__length = 0;
 59 			self.__synchronize = settings.synchronize || false; 
 60 			
 61 			if(typeof(settings.onAllReady) === 'function'){
 62 				self.onAllReady = settings.onAllReady;
 63 			}
 64 		},
 65 		
 66 		/**
 67 		 * Function indicates if all handlers are ready
 68 		 * @function
 69 		 * @name py.Sync#isAllReady
 70 		 * 
 71 		 * @returns {Boolean}
 72 		 */
 73 		isAllReady: function(){
 74 			var registry = this.__registry,
 75 				key;
 76 			
 77 			for(key in registry){
 78 				if(registry.hasOwnProperty(key) && !registry[key].isReady){
 79 					return false; 
 80 				}
 81 			}
 82 			return true;
 83 		},
 84 		
 85 		/**
 86 		 * Function clears the current handlers registry.
 87 		 * Registry should be cleared before adding new set of handlers,
 88 		 * to synchronize.
 89 		 * 
 90 		 * @function
 91 		 * @name py.Sync#clear
 92 		 */
 93 		clear: function(){
 94 			this.__registry = {};
 95 			this.__length = 0;
 96 		},
 97 		
 98 		/**
 99 		 * Function adds handler which should be synchronized
100 		 * to the registry.
101 		 * @function
102 		 * @name py.Sync#addHandler
103 		 * 
104 		 * @param {Object} source an object which contains handler (i.e XMLHttpRequest instance)
105 		 * @param {String} handlerName
106 		 * @param {[Object]} context (optional) to which handler should be applied 
107 		 */
108 		addHandler: function(source, handlerName, context){
109 			var self = this,
110 				handler = source[handlerName],
111 				key,
112 				registered;
113 			
114 			context = context || source;
115 			
116 			if(typeof(handler) === 'function'){
117 				key = createKey.call(self);
118 
119 				source[handlerName] = function(){	
120 					var registryKey,
121 						inSync = self.__synchronize,
122 						registry = self.__registry;
123 					
124 					registered = registry[key];
125 					if(registered){
126 						
127 						registered.isReady = true;
128 						registered.args = arguments;
129 						
130 						if(!inSync){
131 							callRegistered(registered);
132 							delete registry[key];
133 						}
134 						
135 						if(self.isAllReady()){
136 							if(inSync){
137 								for(registryKey in registry){
138 									if(registry.hasOwnProperty(registryKey)){
139 										callRegistered(registry[registryKey]);
140 										delete registry[registryKey];
141 									}
142 								}
143 							}
144 							
145 							if(self.onAllReady){
146 								self.onAllReady();
147 							}
148 						}
149 					
150 					} else {
151 						handler.apply(context || this, arguments);
152 						source[handlerName] = handler;
153 					}
154 					
155 				};
156 				
157 				self.__registry[key] = {
158 					handler: handler,
159 					clearAspect: function(){
160 						source[handlerName] = handler;
161 					},
162 					ctx: context || source,
163 					isReady: false
164 				};
165 			 }
166 		},
167 		/**
168 		 * User defined function which will be called whan all handlers 
169 		 * were executed
170 		 * 
171 		 * @function
172 		 * @name py.Sync#onAllReady
173 		 */
174 		onAllReady: function(){}
175 	}, $NS);
176 	
177 }(py));
