1 (function($NS){
  2 	
  3 	/**
  4 	 * @class  ListOf creates a List of provided type, and 
  5      * prevents object of different type or instance from being added
  6      * 
  7 	 * @name py.ListOf
  8 	 * @extends py.List
  9 	 * 
 10 	 * @param {Function} constructor of chosen type
 11 	 * 
 12 	 * @example
 13 	 * var list = new ListOf(Number);
 14 	 * list.add({}); //will throw exception
 15 	 * 
 16 	 * @constructor
 17 	 */
 18 	$NS.Class('ListOf', { Extends: $NS.List,
 19 		
 20 		initialize: function(of){
 21 			if(typeof(of) === 'function'){
 22 				this.__of = of;
 23 			}else{
 24 				throw "argument should be a constructor";
 25 			}
 26 		},
 27 		
 28 		/**
 29 		 * Function adds element to the list.
 30 		 * @function
 31 		 * @name py.ListOf#add
 32          * @param {Object} element
 33 		 */
 34 		add: function(element){
 35 			var self = this;
 36 			if((element instanceof self.__of) || (element.constructor && element.constructor === self.__of)){
 37 				$NS.List.prototype.add.call(self, element);
 38 			}else{ 
 39 				throw "not allowed type";
 40 			}
 41 		},
 42 		
 43 		/**
 44 		 * Function executes provided method on every 
 45 		 * object on the list. Additional arguments are passed into the
 46 		 * executing method.
 47 		 * 
 48 		 * @function
 49          * @name py.ListOf#execute
 50 		 * 
 51 		 * @example
 52 		 * py.Class('Point',{
 53 		 *     construct: function(x,y){
 54 		 *        this.x = x;
 55 		 *        this.y = y;
 56 		 *     },
 57 		 *     move: function(byX, byY){
 58 		 *        this.x += byX;
 59 		 *        this.y += byY;
 60 		 *     }
 61 		 * }, 'geomery');
 62 		 * 
 63 		 * var points = py.ListOf(geometry.Point);
 64 		 * pointList.add(new geometry.Point(10,10));
 65 		 * pointList.add(new geometry.Point(5,5));
 66 		 * pointList.add(new geometry.Point(7,9));
 67 		 * 
 68 		 * points.execute('move', 4, 5);
 69 		 * 
 70 		 * 
 71          * @param {String} functionName
 72          * @param {Arguments} arguments
 73 		 */
 74 		execute: function(functionName){
 75 			var args = Array.prototype.slice.call(arguments,1),
 76 				iterator = this.iterator(),
 77 				results = [];
 78 			iterator.each(function(elem){
 79 				results.push(elem[functionName].apply(elem, args));
 80 			});
 81 			return results;
 82 		},
 83 		
 84 		/**
 85 		 * Function sets a property of all objects on the list 
 86 		 * with provided value
 87 		 * 
 88 		 * @function
 89 		 * @name py.ListOf#setAll
 90 		 * 
 91 		 * @example
 92 		 * py.Class('Point',{
 93          *     construct: function(x,y){
 94          *        this.x = x;
 95          *        this.y = y;
 96          *     },
 97          *     move: function(byX, byY){
 98          *        this.x += byX;
 99          *        this.y += byY;
100          *     }
101          * }, 'geomery');
102          * 
103          * var points = py.ListOf(geometry.Point);
104          * pointList.add(new geometry.Point(10,10));
105          * pointList.add(new geometry.Point(5,5));
106          * pointList.add(new geometry.Point(7,9));
107          * 
108          * points.setAll('x', 10);
109 		 * 
110 		 * 
111          * @param {String} property
112          * @param {Object} value
113 		 */
114 		setAll: function(property, value){
115 			var iterator = this.iterator();
116 			iterator.each(function(elem){
117 				elem[property] = value;
118 			});
119 		}
120 	}, $NS);
121 	
122 	
123 }(py));
