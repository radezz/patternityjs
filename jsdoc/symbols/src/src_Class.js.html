1 (function($NS, $GLOBAL){
  2     var KeyProperties = {
  3         'Extends': 1,
  4         'Implements' : 1,
  5         'Mixin': 1,
  6         'Static': 1,
  7         'initialize': 1,
  8         'className': 1
  9     },
 10     typeofObject = 'object',
 11     utils = $NS.utils,
 12     createObject = utils.createObject,
 13     isObject = utils.isObject,
 14     isString = utils.isString,
 15     isArray = utils.isArray,
 16     isFunction = utils.isFunction;
 17     
 18     
 19 	/**
 20 	 * Function mixins the definition into the 
 21 	 * constructor prototype. It will not mix in the 'Extend', 'Implements', 'Mixin', 'Static'
 22 	 * properties as they are the 'key' properties used to define the class
 23 	 * 
 24 	 * @function
 25 	 * @name py.Class#mixinProto
 26 	 * 
 27 	 * @param {Function} proto
 28 	 * @param {Object} definition
 29 	 */
 30 	function mixinProto(proto, definition){
 31         var key,
 32             defProperty;
 33         
 34         if(isArray(definition)){
 35 			utils.forEach.call(definition, function(def){
 36 				mixinProto(proto, def);
 37 			});
 38         } else {
 39 			for (key in definition) {
 40 			    if (definition.hasOwnProperty(key)) {
 41                     defProperty = definition[key];
 42                     if(!KeyProperties[key]){
 43                          proto[key] = defProperty;
 44                     }
 45 			    }
 46 		    }
 47             
 48         } 
 49     }
 50     
 51     /**
 52      * Class's helper function. It creates namespace/package and applies object into 
 53      * created namespace
 54      * 
 55      * @function
 56      * @name py.Class#applyToPackage
 57      * 
 58      * @param {String} pckg
 59      * @param {Object} source
 60      * @param {String} name - optional name if source is not a class constructor
 61      */
 62     function applyToPackage(pckg, source, name){
 63         name = name || source.className;
 64         if(isString(pckg)){
 65             utils.createNS([pckg, '.', name].join(''), source);
 66         }else if(isObject(pckg)){
 67             pckg[name] = source;
 68         }else{
 69             $GLOBAL[name] = source;
 70         }
 71     }
 72 
 73 	/**
 74 	 * Function validates input for and returns an object
 75 	 * contanint a namespace and definition, which is used when creating
 76 	 * class.
 77 	 * 
 78 	 * @function
 79 	 * @name py.Class#validateInput
 80 	 * 
 81 	 * @param {String} name
 82 	 * @param {Object} definition
 83 	 * @param {Object | String} package or namespace 
 84 	 * 
 85 	 */
 86 	function validateInput(name, definition, pckg){
 87 		if(!isString(name)){
 88 			throw new TypeError('a class name is required');
 89 		}
 90 		
 91 		if(!isObject(definition)){
 92 		    throw new TypeError(name + ': class definition must be an object');
 93 		}
 94 		
 95 		if(pckg && !isString(pckg) && (typeof(pckg) !== typeofObject)){
 96             throw new TypeError(name + ': package must be an object or string');
 97 		}
 98     }
 99     
100     /**
101      * Function validates if constructor's prototype implements
102      * provided Interface. Is used within class to check if created
103      * Class contains implementation of an previously created Interface
104      * 
105      * @function
106      * @name py.Class#isImplementing
107      * 
108      * @param {Object} construct
109      * @param {Object | Function} implement
110      */
111     function isImplementing(construct, implement, nsOrDefinition){
112         var checkFor = (isFunction(implement))? implement.prototype : implement,
113 			implementsErr;
114         
115         if(!utils.isImplementing(construct.prototype, checkFor)){
116 			implementsErr = nsOrDefinition + " implementation for interface is missing";
117 			throw new Error(implementsErr);
118         }
119     }

121     /**
122      * Class's private helper function. It is used to copy all non primitive
123      * properties which should be cleared on when class is created, to prevent 
124      * prototype static references
125      * 
126      * @private
127      * 
128      * @param {Object} src 
129      * @param {Array} forReinit
130      */
131     function getForReinit(src, forReinit){
132         var key;
133         for(key in src){
134             if(src.hasOwnProperty(key) && src[key] && typeof(src[key]) === 'object'){   
135                 forReinit.push({
136                    key: key,
137                    value: src[key] 
138                 });
139             }
140         }
141             
142     }
143     
144     /**
145      * Class's private helper function which is called with the Class
146      * scope. It is used to reinit no primitive objects 
147      * to its basic state. i.e clear arrays, change objects to empty objects etc
148      * 
149      * @private
150      * 
151      * @param {Array} forReinit
152      */
153     function reinitObjects(forReinit){
154         var i = forReinit.length,
155             self = this,
156             prop;
157          
158          while(i--) {
159             prop = forReinit[i];
160             self[prop.key] = createObject(prop.value);
161          }
162     }

164     /**
165      * @class Base Class creator function. It will create instantiable constructor
166      * of user defined class object.
167      * 
168      * @name py.Class
169      * 
170      * @example
171      * py.Class('MyParentClass', {
172      *    initialize: function(){
173      *         //parent constructor 
174      *    },
175      *    prop: 'value',
176      *    func: function(){
177      *     
178      *    } 
179      *    //...
180      *    //define methods and properties here
181      * }, 'my.package'); //package will be created if not defined
182      * 
183      * 
184      * py.Class('MyClass', { Extends: my.package.MyParentClass
185      *    initialize: function(){
186      *      //use this._parent to access parent functions, and apply to do 'super' calls
187      *      this._parent.initialize.apply(this, arguments);
188      * 
189      *      //place constructor code here 
190      *    } 
191      * }, 'my.package')
192      * 
193      * var myInstance = new my.package.MyClass();
194      * 
195      * @constructor
196      * 
197      * @param {String} name name for the created class
198      * @param {Object} definition definition formed as object
199      * @param {Object | String} pckg optional package a namespace in which the class constructor should be placed
200      * 
201      * @returns {Function} - constructor which can create an defined object
202      */
203     function Class(name, definition, pckg){  
204         var Construct,
205             forReinit = [],
206             Extends,
207             Mixin,
208             Static,
209             Implements;
210         
211         validateInput(name, definition, pckg);
212         
213         Extends = definition.Extends;
214         Mixin = definition.Mixin;
215         Static = definition.Static;
216         Implements = definition.Implements;
217         
218         //gather memeber variables for reinit
219         if(isFunction(Extends)){
220             getForReinit(Extends.prototype, forReinit);
221         }
222         
223         getForReinit(definition, forReinit);
224         
225         if(isFunction(Mixin)){
226             getForReinit(Mixin.prototype, forReinit);
227         } else if(isObject(Mixin)){
228             getForReinit(Mixin.prototype, forReinit);
229         }
230         
231         //create optimized constructor 
232         if (forReinit.length) {  
233             if(isFunction(definition.initialize)){
234                 Construct = function(){
235                     if(forReinit.length){
236                         reinitObjects.call(this, forReinit);    
237                     }
238                     definition.initialize.apply(this, arguments);
239                 };  
240             } else {
241                 Construct = function(){
242                     if(forReinit.length){
243                         reinitObjects.call(this, forReinit);    
244                     }
245                 };
246             }
247         } else {
248             Construct = definition.initialize || function(){};
249         }
250         
251         Construct.className = name;
252         
253         //prototype inheritance
254         if(isFunction(Extends)){
255             Construct.prototype = createObject(Extends.prototype);
256         }
257         //move definition to prototype
258         mixinProto(Construct.prototype, definition);
259         
260         //mix-in Mixin object or Class
261         if(Mixin){
262             if(isFunction(Mixin)){
263                 mixinProto(Construct.prototype, Mixin.prototype);    
264             }else{
265                 mixinProto(Construct.prototype, Mixin);    
266             }
267         }
268         
269         //validate interface implementation
270         if(Implements){
271             isImplementing(Construct, Implements);
272         }
273         
274         //mix-in static declaration to constructor
275         if(Static){
276             utils.mixin(Construct, Static);
277         }
278 		
279 		applyToPackage(pckg, Construct);
280 		
281         return Construct;
282     }
283     
284     Class.prototype = {
285         isImplementing: isImplementing,
286         validateInput: validateInput,
287         mixinProto: mixinProto,
288         applyToPackage: applyToPackage
289     };
290     
291     $NS.Class = Class;
292     
293 }(py, this));
