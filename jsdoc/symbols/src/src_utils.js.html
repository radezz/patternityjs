127 		/**
128 		 * Function extracts object type returned by object.toString.

139 		/**
140 		 * Function creates namespace (if does not exists) and 
141 		 * applies and object to the namespace (if provided).
142 		 * @function
143 		 * @name py.utils#createNS
144 		 * 
145 		 * @param {String} namespace 
146 		 * @param {Object} [optional] object to apply in this namespace
147 		 * 
148 		 * @example
149 		 * var myObj = {};
150 		 * py.utils.createNS('my.name.space.myObj',myObj);
151 		 * 
152 		 * my.name.space.myObj === myObj //will be true
153 		 * 
154 		 */
155 		function createNS(namespace, apply){
156 			var ns = namespace.split('.'),
157 				nsPart = ns.shift(),
158 				targetObject = $Global; //this needs to refer to global
159 			
160 			while(nsPart){
161 				if(!targetObject[nsPart]){
162 					targetObject[nsPart] = {};
163 				}
164 				
165 				if(ns.length === 0 && apply){
166 					targetObject[nsPart] = apply;
167 				}
168 				
169 				targetObject = targetObject[nsPart];	
170 				nsPart = ns.shift();
171 			}
172 			
173 			return targetObject;
174 		}

199 	    /**
200 	     * Function mixins methods and properties from source  object(s)
201 	     * to target object, but does not modify the prototype
202 	     * 
203 	     * @function
204 	     * @name py.utils#mixin
205 	     * 
206          * @param {Object} target
207          * @param {Object | Array} source it can be also array of objects
208 	     */
209 	    function mixin(target, source){
210 	        var key,
211 	            i,
212 	            l;
213 	        if(target && source){
214 				if(isArray(source)){
215 		            for(i=0, l=source.length; i<l; i++){
216 		                mixin(target, source[i]);
217 		            }
218 		        }else{
219 		            for(key in source){
220 		                if(source.hasOwnProperty(key)){
221 		                    target[key] = source[key];
222 		                }
223 		            }
224 		        }
225 		        
226 		        return target;
227 	        }    
228 	    }

230 	    /**
231 	     * Function checks if target object implements provided 
232 	     * interface (object functions and properties)
233 	     * 
234 	     * @function
235 	     * @name py.utils#isImplementing
236 	     * 
237          * @param {Object} target
238          * @param {Object} implement
239 	     */
240 	    function isImplementing(target, implement){
241 			var targetProperty, 
242 				key, 
243 				i, 
244 				l;
245 			
246 			if (isArray(implement)) {
247 				for ( i = 0, l = implement.length; i < l; i++) {
248 					if (!isImplementing(target, implement[i])) {
249 						return false;
250 					}
251 				}
252 				
253 				return true;
254 			
255 			} else if (isObject(implement)) {
256 				for (key in implement) {
257 					if(implement.hasOwnProperty(key)){
258 						targetProperty = target[key];
259 						if (isUndefined(targetProperty) || getType(targetProperty) !== getType(implement[key])) {
260 							return false;
261 						}
262 					}
263 				}
264 
265 				return true;
266 			}
267 	    }

269 	    /**
270 	     * Function applies pair arguments to target function. It is 
271 	     * possible to overload 'key-value' type function to consume 
272 	     * objects as single parameter
273 	     * 
274 	     * @function
275 	     * @name py.utils#pairCall
276 	     * 
277          * @param {Object} fn
278          * @param {Object} pairArgs
279          * @param {Object} context [optional]
280          * 
281          * @returns {Boolean} which states if pairCall is applied
282          * 
283          * @example
284          * 
285          * function set(prop, value){
286          *     if(arguments.length === 1 && !pairCall(set, arguments[0])){
287          *         target[prop] = value;
288          *     }
289          * }
290          * 
291          * //instead of calling 
292          * set('prop', 10);
293          * //you can call
294          * set({
295          *     prop: 10,
296          *     prop2: 20,
297          *     prop3: 30
298          * })
299          * 
300 	     */
301 	    function pairCall(fn, pairArgs, context){
302 	        var key;
303 	            
304 	        if(check.isFunction(fn) && check.isObject(pairArgs)){
305 	            for(key in pairArgs){
306 	                if(pairArgs.hasOwnProperty(key)){
307 	                    fn.call(context, key, pairArgs[key]);
308 	                }
309 	            }
310 	            return true;
311 	        }
312 	        
313 	        return false;
314 	    }
