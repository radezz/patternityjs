162 		/**
163 		 * Function creates namespace (if does not exists) and 
164 		 * applies and object to the namespace (if provided).
165 		 * @function
166 		 * @name py.utils#createNS
167 		 * 
168 		 * @param {String} namespace 
169 		 * @param {Object} [optional] object to apply in this namespace
170 		 * 
171 		 * @example
172 		 * var myObj = {};
173 		 * py.utils.createNS('my.name.space.myObj',myObj);
174 		 * 
175 		 * my.name.space.myObj === myObj //will be true
176 		 * 
177 		 */
178 		function createNS(namespace, apply){
179 			var ns = namespace.split('.'),
180 				nsPart = ns.shift(),
181 				targetObject = $Global; //this needs to refer to global
182 			
183 			while(nsPart){
184 				if(!targetObject[nsPart]){
185 					targetObject[nsPart] = {};
186 				}
187 				
188 				if(ns.length === 0 && apply){
189 					targetObject[nsPart] = apply;
190 				}
191 				
192 				targetObject = targetObject[nsPart];	
193 				nsPart = ns.shift();
194 			}
195 			
196 			return targetObject;
197 		}

232 	     * Function mixins methods and properties from source  object(s)
233 	     * to target object, but does not modify the prototype
234 	     * 
235 	     * @function
236 	     * @name py.utils#mixin
237 	     * 
238          * @param {Object} target
239          * @param {Object | Array} source it can be also array of objects
240 	     */
241 	    function mixin(target, source){
242 	        var key,
243 	            i,
244 	            l;
245 	        if(target && source){
246 				if(isArray(source)){
247 		            for(i=0, l=source.length; i<l; i++){
248 		                mixin(target, source[i]);
249 		            }
250 		        }else{
251 		            for(key in source){
252 		                if(source.hasOwnProperty(key)){
253 		                    target[key] = source[key];
254 		                }
255 		            }
256 		        }
257 		        
258 		        return target;
259 	        }    
260 	    }

262 	    /**
263 	     * Function checks if target object implements provided 
264 	     * interface (object functions and properties)
265 	     * 
266 	     * @function
267 	     * @name py.utils#isImplementing
268 	     * 
269          * @param {Object} target
270          * @param {Object} implement
271 	     */
272 	    function isImplementing(target, implement){
273 			var targetProperty, 
274 				key, 
275 				i, 
276 				l;
277 			
278 			if (isArray(implement)) {
279 				for ( i = 0, l = implement.length; i < l; i++) {
280 					if (!isImplementing(target, implement[i])) {
281 						return false;
282 					}
283 				}
284 				
285 				return true;
286 			
287 			} else if (isObject(implement)) {
288 				for (key in implement) {
289 					if(implement.hasOwnProperty(key)){
290 						targetProperty = target[key];
291 						if (isUndefined(targetProperty) || getType(targetProperty) !== getType(implement[key])) {
292 							return false;
293 						}
294 					}
295 				}
296 
297 				return true;
298 			}
299 	    }

301 	    /**
302 	     * Function applies pair arguments to target function. It is 
303 	     * possible to overload 'key-value' type function to consume 
304 	     * objects as single parameter
305 	     * 
306 	     * @function
307 	     * @name py.utils#pairCall
308 	     * 
309          * @param {Object} fn
310          * @param {Object} pairArgs
311          * @param {Object} context [optional]
312          * 
313          * @returns {Boolean} which states if pairCall is applied
314          * 
315          * @example
316          * 
317          * function set(prop, value){
318          *     if(arguments.length === 1 && !pairCall(set, arguments[0])){
319          *         target[prop] = value;
320          *     }
321          * }
322          * 
323          * //instead of calling 
324          * set('prop', 10);
325          * //you can call
326          * set({
327          *     prop: 10,
328          *     prop2: 20,
329          *     prop3: 30
330          * })
331          * 
332 	     */
333 	    function pairCall(fn, pairArgs, context){
334 	        var key;
335 	            
336 	        if(check.isFunction(fn) && check.isObject(pairArgs)){
337 	            for(key in pairArgs){
338 	                if(pairArgs.hasOwnProperty(key)){
339 	                    fn.call(context, key, pairArgs[key]);
340 	                }
341 	            }
342 	            return true;
343 	        }
344 	        
345 	        return false;
346 	    }

351 	     * @function
352 	     * @name py.utils#forEach
353 	     * 
354 	     * @example
355 	     * py.utils.forEach.call([1,2,3], function(item, index, array){
356 	     *     //...
357 	     * });
358 	     * 
359          * @param {Object} fn
360 	     */
361 	    function forEach(fn) {
362 	        var self = this,
363 	            i,
364 	            l;
365 	            
366 	        if(!check.isArray(this)){
367 	            throw new TypeError("Need an Array as scope");
368 	        }
369 	        
370 	        if(!check.isFunction(fn)) {
371 	            throw new TypeError("Need function");
372 	        }
373 	        
374 	        for(i=0, l=self.length; i<l; i++) {
375 	            fn(self[i], i, self);   
376 	        } 
377 	    }
